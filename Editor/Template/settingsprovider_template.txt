using System.Collections.Generic;
using UnityEditor;
using UnityEngine;
using fsi.prototyping.Spacers;
using UnityEngine.UIElements;

namespace [NAMESPACE].Editor
{
    public static class [NAME]SettingsProvider
    {
        [SettingsProvider]
        public static SettingsProvider CreateSettingsProvider()
        {
            SettingsProvider provider = new("Fantazee/[NAME]", SettingsScope.Project)
                                        {
                                            label = "[NAME]",
                                            activateHandler = OnActivate,
                                        };
            
            return provider;
        }
    
        private static void OnActivate(string searchContext, VisualElement root)
        {
            SerializedObject settingsProp = [Name]Settings.GetSerializedSettings();
            
            Label title = new Header(0, "[NAME] Settings");
            root.Add(title);
            
            root.Add(Spacer.Wide());
            
            root.Bind(settingsProp);
        }
    }
}
